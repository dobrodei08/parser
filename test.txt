	var count = $(obj).val();

	var element_id = "";
	var data_item = $(obj).attr("data-item");

	$(".product_count").each(function(){
		if($(this).val() > 0)
		{
			console.log(count);
			var count_floor = count / data_item;
			if(data_item) {
				if (count < data_item) {
					count = "&quantity[]=" + data_item;
				}
				else if (count > data_item && count_floor / Math.floor(count_floor) == 1) {
					count = "&quantity[]=" + $(this).val();
				}
				else {
					count = Math.ceil(count_floor);
					count = "&quantity[]=" + count * data_item;
				}
				element_id += "&element_id[]="+$(this).parent().find(".product_id_hid").val();
			}
			else{
				count = "&quantity[]="+$(this).val();
				element_id += "&element_id[]="+$(this).parent().find(".product_id_hid").val();
			}
		}
		$(this).val('0');
	});

		type: "POST",
		url: "/bitrix/templates/.default/components/bitrix/catalog.section/line/add_to_basket.php",

         Сегодня классическая (шестиструнная) гитара — самый распространенный вид этого старинного струнного музыкального инструмента. Формирование ее внешнего вида (размеры, количество струн) закончилось только в
         <span lang="EN-US">
          XIX
         </span>
         веке и своим рождением шестиструнная гитара обязана испанскому мастеру Антонио Торресу, который придал ей окончательную внешний вид. С тех пор вид классической гитары остается неизменным.
        </p>
        <p class="MsoNormal" style="text-align: justify; text-indent: 35.3pt;">
         На классической гитаре установлено шесть
         <a href="http://lutner.ru/catalog/struny_dlya_klassicheskikh_gitar/">
          струн
         </a>
         , изготавливаемых из синтетических материалов: нейлона или карбона. К другим отличительным особенностям можно отнести более широкий гриф и двенадцать ладов на грифе до края корпуса (вместо 14 на всех остальных гитарах).
        </p>
        <p class="MsoNormal" style="text-align: justify; text-indent: 35.3pt;">
         Для изготовления классических гитар используется разные породы дерева: для задней деки  и обечаек  берут красное дерево или палисандр; для верхней деки – кедр или ель; гриф может изготавливаться как из кедра, так и из красного дерева. В классических гитарах начального уровня в основном используется ламинат - облагороженная шпоном дорогих пород дерева фанера. Это позволяет сохранить невысокую стоимость инструмента и делает его более простым в эксплуатации.
         <br/>
        </p>
        <p class="MsoNormal" style="text-align: justify; text-indent: 35.3pt;">
         Перебор является главной техникой извлечения звука на классической гитаре. Для усиления яркости и акцентирования звучания музыканты используют ногти, а не привычные многим медиаторы, которые могут повредить корпус гитары. Техника игры, также, имеет свои особенности. В первую очередь – это классическая посадка, при которой гитара выемкой обечайки располагается на левой ноге гитариста, стоящей на специальной
         <a href="http://lutner.ru/catalog/postavki_dlya_nogi/">
          подставке
         </a>
         <span>
          . Гриф гитары смотрит вверх под углом примерно в 45
          <sup>
           о
          </sup>
          . Сегодня среди различных гитарных аксессуаров можно найти «
         </span>
         <a href="http://lutner.ru/catalog/postavki_dlya_nogi/">
          суппорты
         </a>
         » - подставки для классической гитары, устанавливаемые на колено. При использовании такого «суппорта» отпадает необходимость в подставке под ногу.
        </p>
